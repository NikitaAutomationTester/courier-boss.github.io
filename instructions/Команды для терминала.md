1. Просмотр содержимого текущей директории

ls -la # просмотр всех файлов в текущей дериктории

2. Переходы по директории проекта

cd /root          # зайти в папку root
cd BBEx           # зайти в папку BBEx (из текущей папки)
cd ..             # выйти на уровень назад (как "Назад" в Проводнике)
cd ~              # зайти в домашнюю папку
cd /              # зайти в корень системы
cd -              # вернуться в предыдущую папку (как стрелка "Назад")
